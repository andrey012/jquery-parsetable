jQuery ParseTable plugin demo page

Simply copy (Ctrl-C) few cells from your spreadsheet or table in text application and click "test link"

    jQuery(document).ready(function() {
        jQuery("#test").parseTable({
            success: function(data){alert('the action was completed, here is the data \n' + data.join("\n"));},
            cancel: function(){alert('the action was cancelled');},
        });
    });
</script>
